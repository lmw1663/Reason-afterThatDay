```sql
-- C-SSRS 응답 + 위기 이벤트 추적
create table public.crisis_assessments (
  id uuid primary key default gen_random_uuid(),
  user_id uuid not null references public.users(id) on delete cascade,
  assessed_at timestamptz not null default now(),
  source text not null check (source in ('onboarding', 'graduation', 'modal_trigger', 'periodic')),
  -- C-SSRS 6항 응답 (boolean 또는 integer 0~5)
  q1_passive_ideation boolean not null default false,
  q2_active_ideation boolean not null default false,
  q3_method boolean not null default false,
  q4_intent boolean not null default false,
  q5_plan boolean not null default false,
  q6_recent_attempt boolean not null default false,
  -- 산출 등급
  severity text not null check (severity in ('safe', 'caution', 'high', 'urgent')),
  -- 처치 액션 기록
  actions_triggered text[] not null default '{}',  -- e.g. {graduation_locked, hotline_shown, redial_pushed}
  followup_due_at timestamptz,
  followup_completed_at timestamptz,
  created_at timestamptz default now()
);

-- 앱 안전 잠금 상태 (한 사용자 1행)
create table public.safety_lockouts (
  user_id uuid primary key references public.users(id) on delete cascade,
  locked_at timestamptz not null default now(),
  reason text not null,  -- 'crisis_assessment', 'manual'
  graduation_locked boolean not null default false,
  decision_locked boolean not null default false,
  expires_at timestamptz,  -- null = 무기한, 사용자 또는 followup 완료 시 해제
  released_at timestamptz,
  released_by text  -- 'user_acknowledgment' | 'followup_resolution'
);

-- RLS
alter table public.crisis_assessments enable row level security;
alter table public.safety_lockouts enable row level security;

create policy "users_own_crisis" on public.crisis_assessments
  for all using (auth.uid() = user_id) with check (auth.uid() = user_id);

create policy "users_own_lockout" on public.safety_lockouts
  for all using (auth.uid() = user_id) with check (auth.uid() = user_id);

-- 의료정보 보존 정책 — 5년 후 자동 익명화
-- (cron job은 별도 구현, 19_data_retention.sql)
```

```typescript
const PERSONAS: PersonaCode[] = ['P01', 'P02', /* ... */ 'P20'];

// 페르소나-분류체계.md §4 점수 부여 규칙을 그대로 옮긴 표
export const SCORING_RULES: Record<string, Partial<Record<PersonaCode, number>>> = {
  'q1.ghosted': { P04: 3, P11: 1, P20: 1 },
  'q2.too_sensitive': { P01: 2 },           // 검증 §2-2 반영(이전 +3)
  'q2.too_sensitive+q6.yes': { P01: 5 },    // 두 응답 동시 양성
  'q2.alone_better': { P02: 3, P12: 1 },
  // ... (전체 표는 페르소나-분류체계.md §4와 1:1 동기화 — 단일 진실 소스 유지 위해 JSON으로 분리하는 것도 검토)
};

export function classifyPersona(
  responses: OnboardingResponses,
  axes: PsychAxes
): PersonaResult {
  // 1. 우선 분기 (검증 §2-1, §2-6 반영)
  if (axes.a8_crisis >= 3) return { mode: 'crisis_lockout', primary: null, secondary: null };
  if (responses.bereavement === true) return { mode: 'bereavement_referral', primary: null, secondary: null };
  if (responses.userInfidelity === true) {
    // P14는 강제 추가, 나머지 점수 산출 후 부 페르소나 결정
  }

  // 2. 점수 합산
  const scores: Record<PersonaCode, number> = Object.fromEntries(PERSONAS.map(p => [p, 0])) as any;
  for (const [key, deltas] of matchedRules(responses, axes)) {
    for (const [persona, delta] of Object.entries(deltas)) {
      scores[persona as PersonaCode] += delta;
    }
  }

  // 3. 기본값 처리 (검증 §2-9 변별 타당도 보강 — 점수차 임계 강화)
  const sorted = Object.entries(scores).sort((a, b) => b[1] - a[1]);
  const [topPersona, topScore] = sorted[0];
  if (topScore < 5) return { mode: 'standard', primary: 'P12', secondary: null, scores };

  const [secondPersona, secondScore] = sorted[1];
  const secondary = (topScore - secondScore >= 2) ? secondPersona : null;

  // 4. 외부복잡도 강제 부 페르소나 (3부 알고리즘 6단계)
  let forcedSecondary = null;
  if (axes.a6_complexity >= 2) {
    forcedSecondary = pickComplexityPersona(responses);  // P15/P16/P18 중
  }

  return {
    mode: 'standard',
    primary: topPersona,
    secondary: forcedSecondary || secondary,
    scores
  };
}
```

```typescript
    `현재 사용자 상태(라벨 노출 금지): 톤 — ${block.tone}`,

    `권장 프레이밍: ${block.recommendedFraming}`,

    DISCLAIMER_FOOTER,
  ].join('\n\n');

// 응답 후처리: 금지 키워드 lint
export function lintResponse(response: string, persona: PersonaCode): { ok: boolean; violations: string[] } {
  const block = PERSONA_SYSTEM_BLOCKS[persona];
  const violations = block.forbiddenKeywords.filter(k => response.includes(k));
  return { ok: violations.length === 0, violations };
}
```
